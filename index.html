Que1: Find the Smaller Angle

Ans:

var Minimal_Angle = (h, m) => 
{
    let hourAngle = (h % 12) * 30 + m / 2;
    let minuteAngle = m * 6;
    let angleDiff = Math.abs(hourAngle - minuteAngle);
    let angle = Math.min(angleDiff, 360 - angleDiff);
    return angle;   
};

Que3: Perfect Number Check.

Ans:

var Perfect_Check = (N) => 
{
 let temp = 0;
 for(let i = 0; i < N/2; i++){
   if(N%i === 0)
   {
     temp += i;
   }
 }
 if(temp === N && temp !== 0){
   return "YES";
 }
 return "NO";
};

Que4: Reverse a Number.

Ans:

var Reverse_Number = (N) => 
{
   let newNum = 0
  let temp = 0;
  while(N > 0){
    temp = N%10;
    newNum = newNum*10+ temp;
    N = Math.floor(N/10);
  }
  return newNum;
};
